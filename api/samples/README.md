```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from bs4 import BeautifulSoup
import time
import re  # 정규표현식 모듈 import

# 크롬 웹 드라이버의 경로를 지정합니다.
driver = webdriver.Chrome()

urls = [
    'https://ggaggamukja.com/product/%EA%B0%95%EC%95%84%EC%A7%80-%EC%A0%84%EC%9A%A9-%EB%A9%8D%EB%A7%A5%EC%A3%BC-220ml/106/category/75/display/1/',
        ]  # URL 리스트

for url in urls:
    driver.get(url)
    time.sleep(5)  # 각 요청 사이에 5초의 대기 시간을 둡니다.

    # 섬네일 이미지를 모두 가져옵니다.
    thumbnails = driver.find_elements(By.CSS_SELECTOR, 'li.xans-record- > img.ThumbImage')

    big_images = []  # 배열 선언
    # 각 섬네일 이미지를 클릭하고, 변경되는 큰 이미지의 주소를 크롤링합니다.
    for thumbnail in thumbnails:
        if thumbnail.is_displayed():  # 웹 요소가 보이는지 확인합니다.
            wait = WebDriverWait(driver, 10)  # 웹 드라이버를 최대 10초간 대기하도록 설정합니다.
            element = wait.until(EC.element_to_be_clickable(
                (By.CSS_SELECTOR, 'li.xans-record- > img.ThumbImage')))  # 웹 요소가 클릭 가능한 상태가 될 때까지 대기합니다.
            element.click()
            time.sleep(1)  # 이미지가 로드되는 동안 대기합니다.
            big_image = driver.find_element(By.CSS_SELECTOR, 'img.BigImage')
            big_images.append(big_image.get_attribute('src'))  # 배열에 추가

    # Selenium을 통해 페이지의 HTML을 가져와 BeautifulSoup 객체를 생성합니다.
    soup = BeautifulSoup(driver.page_source, "html.parser")

    # 상품명
    product_name = soup.find("div", {"class": "headingArea"}).find("h2").text

    # 가격
    price = soup.find("strong", {"id": "span_product_price_text"}).text
    price = re.sub(r'\D', '', price)  # 숫자만 추출

    # 옵션
    option_groups = soup.find_all("optgroup")
    options_list = []
    if option_groups:
        for group in option_groups:
            label = group['label']
            options = group.find_all("option")
            for option in options:
                if 'selected' not in option.attrs and 'disabled' not in option.attrs:
                    options_list.append({label: option['value']})
    else:
        options = soup.find_all("option")
        options_list = [option['value'] for option in options if
                        'selected' not in option.attrs and 'disabled' not in option.attrs]

    # 섬네일 이미지 주소
    thumbnails = soup.find_all("img", {"class": "ThumbImage", "link_image":""})
    thumbnails = [thumbnail['src'] for thumbnail in thumbnails]  # 배열로 변경

    # 상세 이미지 주소
    detail_images_divs = soup.find_all("div", {"class": "edb-img-tag-w"})
    detail_images = []
    for div in detail_images_divs:
        img = div.find("img")
        if img is not None:
            detail_images.append(
                '//ggaggamukja.com' + img.get('ec-data-src', ''))  # img가 None이 아니라면 ec-data-src를 가져옵니다.

    # 출력
    print(f"""
        {{
          _id: await nextSeq('product'),
          seller_id: 2,
          price: {price},
          shippingFees: 0,
          show: true,
          active: true,
          name: '{product_name}',
          options: {options_list},
          mainImages: {big_images},
          detailImages: {thumbnails},
          descriptImages: {detail_images},
          content: `
                  <div class="product-detail">
                      <p>{product_name}</p>
                  </div>`,
          createdAt: getTime(-41, -60 * 60 * 2),
          updatedAt: getTime(-40, -60 * 15),
          extra: {{
            isNew: true,
            isBest: true,
            category: ['PC-party'],
            quantity: 100,
            buyQuantity: 0,
            order: 0,
          }},
        }},
    """)

driver.quit()
```
